def verificar_identificacion(identificacion):
    # Cargar los datos del archivo JSON de clientes
    with open("ClientesDataBase.json", "r") as file:
        clientes = json.load(file)
    
    # Buscar el cliente por identificación
    for cliente in clientes["Clientes"]:
        if cliente["Identificacion"] == identificacion:
            return cliente["Nombre"]
    
    # Si no se encuentra la identificación del cliente
    return None
